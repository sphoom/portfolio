data_preprocessing.py 

import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler

def preprocess_data(df):
    """
    Preprocess the data for modeling
    
    Parameters:
    df (pandas.DataFrame): Raw input data
    
    Returns:
    pandas.DataFrame: Processed data ready for modeling
    """
    # Handle missing values
    df = df.fillna(df.mean())
    
    # Create new features
    df['feature_ratio'] = df['feature_1'] / df['feature_2']
    
    # Scale numerical features
    scaler = StandardScaler()
    numeric_cols = df.select_dtypes(include=[np.number]).columns
    df[numeric_cols] = scaler.fit_transform(df[numeric_cols])
    
    return df

data_extraction.sql 

-- Extract relevant data from transaction table
SELECT 
    t.customer_id,
    c.customer_segment,
    t.transaction_date,
    SUM(t.amount) AS total_amount,
    COUNT(t.transaction_id) AS transaction_count,
    AVG(t.amount) AS average_transaction_value
FROM 
    transactions t
JOIN 
    customers c ON t.customer_id = c.customer_id
WHERE 
    t.transaction_date BETWEEN '2024-01-01' AND '2024-12-31'
    AND t.status = 'completed'
GROUP BY 
    t.customer_id,
    c.customer_segment,
    t.transaction_date
ORDER BY 
    t.transaction_date;
